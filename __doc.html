func BaseCodePathGithub

func BaseCodePathGithub(gitHubName string, subDirNames ...string) string

Returns the *filepath*-joined full directory path for a specified $GOPATH/src/github.com sub-directory. Example: util.BaseCodePathGo("metaleap", "go-util") = "c:\gd\src\github.com\metaleap\go-util" if $GOPATH is c:\gd.

func BaseCodePathGo

func BaseCodePathGo(subDirNames ...string) string

Returns the *filepath*-joined full directory path for a specified $GOPATH/src sub-directory. Example: util.BaseCodePathGo("tools", "importers", "sql") = "c:\gd\src\tools\importers\sql" if $GOPATH is c:\gd.

func ParseVersion

func ParseVersion(verstr string) (majorMinor []int, both float64)

Attempts to extract major and minor version components from a string that begins with a version number. Example: returns []int{3, 2} and float64(3.2) for a verstr that is "3.2.0 - Build 8.15.10.2761"
